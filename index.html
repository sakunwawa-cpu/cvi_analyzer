            const el = document.getElementById(inputId);

            let val = parseInt(el.value) || 0;

            val += delta;

            if (inputId === 'numExperts') { if (val < 3) val = 3; if (val > 20) val = 20; }

            else if (inputId === 'numItems') { if (val < 1) val = 1; if (val > 100) val = 100; }

            el.value = val;

            if (inputId === 'numItems' && isDetailsEnabled) renderItemInputs();

        }

        function setScore(item, expert, score) {

            const key = `${item}-${expert}`;

            scores[key] = score;

            for (let s = 1; s <= 4; s++) {

                const btn = document.getElementById(`btn-${item}-${expert}-${s}`);

                btn.classList.remove('selected', 'bg-red-500', 'bg-orange-500', 'bg-green-500', 'bg-emerald-600', 'text-white');

                btn.classList.add('bg-slate-100', 'text-slate-400');

            }

            const selectedBtn = document.getElementById(`btn-${item}-${expert}-${score}`);

            selectedBtn.classList.remove('bg-slate-100', 'text-slate-400');

            selectedBtn.classList.add('selected', 'text-white');

            if (score === 1) selectedBtn.classList.add('bg-red-500');

            else if (score === 2) selectedBtn.classList.add('bg-orange-500');

            else if (score === 3) selectedBtn.classList.add('bg-green-500');

            else selectedBtn.classList.add('bg-emerald-600');

        }

        function calculateCVI() {

            const totalCells = numItems * numExperts;

            const filledCells = Object.keys(scores).length;

            if (filledCells < totalCells) { showToast(`⚠️ กรุณากรอกคะแนนให้ครบทุกช่อง`); return; }

            

            analysisResults = [];

            let totalICVI = 0;

            let itemsWithFullAgreement = 0;

            

            for (let i = 1; i <= numItems; i++) {

                let relevantCount = 0;

                const expertScores = [];

                for (let e = 1; e <= numExperts; e++) {

                    const score = scores[`${i}-${e}`];

                    expertScores.push(score);

                    if (score >= 3) relevantCount++;

                }

                const iCVI = relevantCount / numExperts;

                if (relevantCount === numExperts) itemsWithFullAgreement++;

                totalICVI += iCVI;

                

                let threshold = numExperts <= 5 ? 1.0 : 0.78;

                let status = iCVI >= threshold ? 'ผ่านเกณฑ์' : 'ต้องปรับปรุง';

                let statusClass = status === 'ผ่านเกณฑ์' ? 'bg-emerald-100 text-emerald-700' : 'bg-rose-100 text-rose-700';

                

                analysisResults.push({ item: i, detail: itemDetails[i] || '', scores: expertScores, relevantCount, iCVI, status, statusClass });

            }

            

            const sCVIAve = totalICVI / numItems;

            const sCVIUA = itemsWithFullAgreement / numItems;

            displaySummary(sCVIAve, sCVIUA);

        }


        function displaySummary(sCVIAve, sCVIUA) {

            document.getElementById('scviAve').textContent = sCVIAve.toFixed(3);

            document.getElementById('scviUa').textContent = sCVIUA.toFixed(3);

            document.getElementById('footerAve').textContent = sCVIAve.toFixed(3);

            document.getElementById('footerUa').textContent = sCVIUA.toFixed(3);

            

            const statusH4 = document.getElementById('overallStatus');

            const statusP = document.getElementById('statusNote');

            

            if(sCVIAve >= 0.90) {

                statusH4.textContent = 'ผ่านเกณฑ์ดีเยี่ยม';

                statusP.textContent = 'เครื่องมือมีความตรงเชิงเนื้อหาสูง';

            } else if(sCVIAve >= 0.80) {

                statusH4.textContent = 'ผ่านเกณฑ์';

                statusP.textContent = 'เครื่องมือมีความตรงเชิงเนื้อหาในเกณฑ์ที่ยอมรับได้';

            } else {

                statusH4.textContent = 'ต้องพิจารณาปรับปรุง';

                statusP.textContent = 'เครื่องมือมีความตรงเชิงเนื้อหาต่ำกว่าเกณฑ์มาตรฐาน';

            }


            renderResultsTable(sCVIAve, sCVIUA);

            document.getElementById('resultsSection').classList.remove('hidden');

            document.getElementById('resultsSection').scrollIntoView();

            showToast('คำนวณผลเรียบร้อย');

            lucide.createIcons();

        }
